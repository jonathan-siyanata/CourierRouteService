CREATE TABLE `Client` (

	  `CLIENT_ID` bigint NOT NULL AUTO_INCREMENT,
	  `FIRST_NAME` varchar(255) NOT NULL,
	  `LAST_NAME` varchar(255)NOT NULL,
	  `DATE_OF_BIRTH` date NOT NULL,
	  `EMAIL_ADDRESS` varchar(255) NOT NULL,
	  `PHONE_NUMBER` varchar(255) NOT NULL,
	  `CLIENT_ID_NUMBER` varchar(255) NOT NULL,
	  `ADDRESS_ID` bigint NOT NULL,
	  PRIMARY KEY (`CLIENT_ID`),
	  KEY `Client_FK` (`ADDRESS_ID`),
	  CONSTRAINT `Client_FK` FOREIGN KEY (`ADDRESS_ID`) REFERENCES `Address` (`ADDRESS_ID`)
) ;

CREATE TABLE `Address` (

	  `ADDRESS_ID` bigint NOT NULL AUTO_INCREMENT,
	  `LAT` double NOT NULL,
	  `LNG` double NOT NULL,
	  `ADDRESS` varchar(255) NOT NULL,
	  PRIMARY KEY (`ADDRESS_ID`)
);

CREATE TABLE `AlgorithmType` (

	  `ALGORITHM_TYPE_ID` bigint NOT NULL AUTO_INCREMENT,
	  `ALGORITHM_TYPE` varchar(255) NOT NULL,
	  PRIMARY KEY (`ALGORITHM_TYPE_ID`)
);

CREATE TABLE `Driver` (

	  `DRIVER_ID` bigint NOT NULL AUTO_INCREMENT,
	  `FIRST_NAME` varchar(255) NOT NULL,
	  `LAST_NAME` varchar(255) NOT NULL,
	  `DATE_OF_BIRTH` date NOT NULL,
	  `EMAIL_ADDRESS` varchar(255) NOT NULL,
	  `PHONE_NUMBER` varchar(255) NOT NULL,
	  `DRIVER_ID_NUMBER` varchar(255) NOT NULL,
	  `PASSWORD` varchar(255) NOT NULL,
	  PRIMARY KEY (`DRIVER_ID`)
);

CREATE TABLE `Product` (

	  `PRODUCT_ID` bigint NOT NULL AUTO_INCREMENT,
	  `PRODUCT_QUANTITY` bigint NOT NULL,
	  `PRODUCT_TYPE` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci NOT NULL,
	  PRIMARY KEY (`PRODUCT_ID`),
	  UNIQUE KEY `Product_Constraint` (`PRODUCT_ID`,`PRODUCT_QUANTITY`,`PRODUCT_TYPE`)
);

CREATE TABLE `Delivery` (

	  `DELIVERY_ID` bigint NOT NULL AUTO_INCREMENT,
	  `DRIVER_ID_NUMBER` varchar(255) NOT NULL,
	  `DELIVERY_DATE` date NOT NULL,
	  `DELIVERY_TIME` time NOT NULL,
	  `PRODUCT_TYPE` varchar(255) NOT NULL,
	  `PRODUCT_QUANTITY` bigint NOT NULL,
	  `DRIVER_ID` bigint NOT NULL,
	  `ROUTE_ID` bigint NOT NULL,
	  `PRODUCT_ID` bigint NOT NULL,
	  `ADDRESS_ID` bigint NOT NULL,
	  `ADDRESS` varchar(255) NOT NULL,
	  PRIMARY KEY (`DELIVERY_ID`),
	  KEY `Delivery_FK` (`DRIVER_ID`),
	  KEY `Delivery_FK_1` (`ROUTE_ID`),
	  KEY `Delivery_FK_3` (`ADDRESS_ID`),
	  KEY `Delivery_FK_2` (`PRODUCT_ID`),
	  CONSTRAINT `Delivery_FK` FOREIGN KEY (`DRIVER_ID`) REFERENCES `Driver` (`DRIVER_ID`),
	  CONSTRAINT `Delivery_FK_1` FOREIGN KEY (`ROUTE_ID`) REFERENCES `Route` (`ROUTE_ID`),
	  CONSTRAINT `Delivery_FK_2` FOREIGN KEY (`PRODUCT_ID`) REFERENCES `Product` (`PRODUCT_ID`),
	  CONSTRAINT `Delivery_FK_3` FOREIGN KEY (`ADDRESS_ID`) REFERENCES `Address` (`ADDRESS_ID`)
);
